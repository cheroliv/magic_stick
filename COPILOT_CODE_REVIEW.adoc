=== 1.3 Architecture Observations

==== Excellent Decisions ✅

1. **Multi-bootloader support** (BIOS + UEFI)
- Handles legacy and modern firmware
- GRUB configuration explicit and safe

2. **Persistence configuration**
- Union mount strategy preserves user data
- Automatically mounted on boot

3. **Atomic update mechanism**
- Detects active partition (mount state + GRUB flags)
- Updates inactive partition
- Flips boot flag (single operation = atomic)
- Automatic rollback if boot fails

4. **CLI image mirrors ISO stack**
- Comment on line 2 explicitly notes this: "Reproduit la stack logicielle CLI..."

==== Dockerfile.base (New)

[source,dockerfile]
----
# docker/Dockerfile.base
# ============================================================
# Magic Stick — Shared Base Layer
# ============================================================
# This stage contains ALL common dependencies shared between:
# - ISO builder (live-build chroot)
# - CLI container (GitHub Actions)
#
# Build once, reuse everywhere.

FROM ubuntu:24.04 as base

ENV DEBIAN_FRONTEND=noninteractive
ENV LC_ALL=en_US.UTF-8
ENV LANG=en_US.UTF-8

# === Versions pinned for reproducibility ===
ARG PYTHON_VERSION=3.14.4
ARG UV_VERSION=0.11.8
ARG NVM_VERSION=v0.40.4
ARG NODE_VERSION=22
ARG PNPM_VERSION=10.33.2
ARG LAZYGIT_VERSION=0.61.1
ARG XH_VERSION=0.25.3
ARG JUST_VERSION=1.50.0

# === Stage 1: System packages (common) ===
RUN apt-get update && apt-get install -y --no-install-recommends \
    ca-certificates \
    curl \
    wget \
    git \
    jq \
    ripgrep \
    fd-find \
    fzf \
    htop \
    nano \
    vim \
    less \
    bash-completion \
    unzip \
    zip \
    p7zip-full \
    pciutils \
    usbutils \
    lsb-release \
    lshw \
    dmidecode \
    net-tools \
    iproute2 \
    bridge-utils \
    dnsutils \
    nmap \
    iperf3 \
    tcpdump \
    ethtool \
    socat \
    screen \
    tmux \
    zsh \
    ffmpeg \
    xz-utils \
    zstd \
    build-essential \
    libssl-dev \
    zlib1g-dev \
    libbz2-dev \
    libreadline-dev \
    libsqlite3-dev \
    llvm \
    libncursesw5-dev \
    tk-dev \
    libxml2-dev \
    libxmlsec1-dev \
    libffi-dev \
    liblzma-dev \
    && rm -rf /var/lib/apt/lists/*

# === Stage 2: Python 3.14.4 + uv ===
RUN curl -fsSL https://pyenv.run | bash

ENV PATH="/root/.pyenv/shims:/root/.pyenv/bin:$PATH"

RUN bash -c 'eval "$(pyenv init -)" && pyenv install ${PYTHON_VERSION}' \
    && ~/.pyenv/versions/${PYTHON_VERSION}/bin/python3 -m pip install --upgrade pip

RUN curl -fsSL https://github.com/astral-sh/uv/releases/download/${UV_VERSION}/uv-x86_64-unknown-linux-gnu.tar.gz \
    | tar -xz -C /usr/local/bin uv

# === Stage 3: Node.js + NVM + pnpm ===
ENV NVM_DIR="/root/.nvm"
ENV NODE_VERSION=22

RUN curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/${NVM_VERSION}/install.sh | bash \
    && . "$NVM_DIR/nvm.sh" \
    && nvm install ${NODE_VERSION} \
    && nvm use ${NODE_VERSION} \
    && nvm alias default ${NODE_VERSION} \
    && npm install -g pnpm@${PNPM_VERSION}

ENV PATH="$NVM_DIR/versions/node/v${NODE_VERSION}/bin:$PATH"

# === Stage 4: Java + SDKMAN ===
RUN curl -s "https://get.sdkman.io" | bash

ENV SDKMAN_DIR="/root/.sdkman"
ENV PATH="$SDKMAN_DIR/candidates/java/current/bin:$PATH"
ENV JAVA_HOME="$SDKMAN_DIR/candidates/java/current"

RUN bash -c 'source "$SDKMAN_DIR/bin/sdkman-init.sh" && \
    sdk install java 25-tem && \
    sdk default java 25-tem'

# === Stage 5: CLI Tools (common) ===
RUN curl -fsSL -o /tmp/lazygit.tar.gz \
    "https://github.com/jesseduffield/lazygit/releases/download/v${LAZYGIT_VERSION}/lazygit_${LAZYGIT_VERSION}_linux_x86_64.tar.gz" \
    && tar -xzf /tmp/lazygit.tar.gz -C /usr/local/bin lazygit \
    && rm /tmp/lazygit.tar.gz

RUN curl -fsSL -o /tmp/xh.tar.gz \
    "https://github.com/ducaale/xh/releases/download/v${XH_VERSION}/xh-v${XH_VERSION}-x86_64-unknown-linux-musl.tar.gz" \
    && tar -xzf /tmp/xh.tar.gz -C /usr/local/bin \
    && rm /tmp/xh.tar.gz

RUN curl -fsSL -o /tmp/just.tar.gz \
    "https://github.com/casey/just/releases/download/${JUST_VERSION}/just-${JUST_VERSION}-x86_64-unknown-linux-musl.tar.gz" \
    && tar -xzf /tmp/just.tar.gz -C /usr/local/bin just \
    && rm /tmp/just.tar.gz

RUN curl -sS https://starship.rs/install.sh | sh -s -- -y

# === Stage 6: Ollama ===
RUN curl -fsSL https://ollama.com/install.sh | sh

# === Stage 7: Flatpak ===
RUN apt-get update && apt-get install -y --no-install-recommends flatpak \
    && rm -rf /var/lib/apt/lists/*

# === Final cleanup ===
RUN apt-get clean && rm -rf /var/lib/apt/lists/* /tmp/* /var/tmp/*

WORKDIR /workspace

ENV SHELL=/bin/zsh
ENV TERM=xterm-256color

==== build.gradle.kts (Updated)

[source,kotlin]
----
// ============================================================
// Docker Images — Multi-Stage Build
// ============================================================

val magicStickVersion = rootProject.file("VERSION").readText().trim()
val baseImage = "magic-stick:base"
val isoBuilderImage = "magic-stick:builder"
val cliImage = "magic-stick-cli:$magicStickVersion"

// Step 1: Build shared base
tasks.register<Exec>("dockerBuildBase") {
    group = "docker"
    description = "Build shared base image (magic-stick:base) — all common dependencies"

    commandLine("docker", "build",
        "--file", "docker/Dockerfile.base",
        "--tag", baseImage,
        ".")

    doFirst {
        logger.info("Building shared base image...")
    }
}

// Step 2: Build ISO builder (depends on base)
tasks.register<Exec>("dockerBuildIsoBuilder") {
    group = "docker"
    description = "Build ISO builder image (depends on base)"
    dependsOn("dockerBuildBase")

    commandLine("docker", "build",
        "--file", "docker/Dockerfile.iso-builder",
        "--build-arg", "BASE_IMAGE=$baseImage",
        "--tag", isoBuilderImage,
        ".")

    doFirst {
        logger.info("Building ISO builder image (from base)...")
    }
}

// Step 3: Build CLI image (depends on base)
tasks.register<Exec>("dockerBuildCli") {
    group = "docker"
    description = "Build CLI image (depends on base)"
    dependsOn("dockerBuildBase")

    val creds = parseDockerhubCreds()
    val repo = if (!creds?.first.isNullOrEmpty()) "${creds?.first}/magic-stick-cli" else "cheroliv/magic-stick-cli"

    commandLine("docker", "buildx", "build",
        "--file", "docker/magic-stick-cli/Dockerfile",
        "--build-arg", "BASE_IMAGE=$baseImage",
        "--tag", "${repo}:${magicStickVersion}",
        "--tag", "${repo}:latest",
        ".")

    doFirst {
        logger.info("Building CLI image (from base)...")
    }
}

// Step 4: Build all images
tasks.register("dockerBuildAll") {
    group = "docker"
    description = "Build all Docker images (base → iso-builder, cli)"
    dependsOn("dockerBuildBase", "dockerBuildIsoBuilder", "dockerBuildCli")

    doLast {
        logger.info("✅ All Docker images built successfully")
    }
}

// Update ISO build to use the new builder image
tasks.named<Exec>("isoBuild") {
    dependsOn("dockerBuildIsoBuilder")  // Changed from "dockerBuild"
}
